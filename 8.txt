function toggleVuln(header) {
  const body = header.nextElementSibling;
  const arrow = header.querySelector('.vuln-toggle');
  const open = body.classList.toggle('open');
  arrow.textContent = open ? '▲' : '▼';
}

// ── Copy code
function copyCode(btn) {
  const code = btn.closest('.code-wrap').querySelector('.code-body pre');
  const text = code ? code.innerText : btn.closest('.code-wrap').querySelector('.code-body').innerText;
  navigator.clipboard.writeText(text).then(() => {
    btn.textContent = 'Copied!';
    setTimeout(() => btn.textContent = 'Copy', 2000);
  });
}
